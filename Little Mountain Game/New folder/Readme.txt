mouse moveing: 
	straight up/down: pitch up/down
	left/right: roll left/right
	right/left click: go faster/slower (will stop and prevent going back)

arrow key:
	up/down: pitch up/down
	left/right: roll left/right

keyboard:
	w: go faster
	s: go slower (WILL go slower)
	f/s: resolution
	+/-: sea level
